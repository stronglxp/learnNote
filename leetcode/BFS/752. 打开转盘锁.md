#### 752. 打开转盘锁

链接：https://leetcode-cn.com/problems/open-the-lock/

```java
public class Test_752 {
    // 转盘向上拨动一次
    private String upOne(String str, int index) {
        char[] chs = str.toCharArray();
        if (chs[index] == '9') {
            chs[index] = '0';
        } else {
            chs[index] += 1;
        }

        return new String(chs);
    }

    // 转盘向下拨动一次
    private String downOne(String str, int index) {
        char[] chs = str.toCharArray();
        if (chs[index] == '0') {
            chs[index] = '9';
        } else {
            chs[index] -= 1;
        }

        return new String(chs);
    }

    public int openLock(String[] deadends, String target) {
        // 死亡数字
        Set<String> deadLock = new HashSet<>();
        // 已经转过的数字
        Set<String> visited = new HashSet<>();
        for (String str : deadends) {
            deadLock.add(str);
        }

        Queue<String> res = new LinkedList<>();
        res.offer("0000");
        visited.add("0000");
        int step = 0;

        while (!res.isEmpty()) {
            int len = res.size();

            // 当前队列中的所有节点向周围扩散
            for (int i = 0; i < len; i++) {
                String cur = res.poll();
                // 如果是死亡数字，则退出这次循环
                if (deadLock.contains(cur)) {
                    continue;
                }

                // 转到target，直接返回step即为最小旋转次数
                if (cur.equals(target)) {
                    return step;
                }

                // 对于当前节点，每个拨轮进行向上和向下转动
                for (int j = 0; j < 4; j++) {
                    String next = upOne(cur, j);
                    if (!visited.contains(next)) {
                        res.offer(next);
                        visited.add(next);
                    }

                    next = downOne(cur, j);
                    if (!visited.contains(next)) {
                        res.offer(next);
                        visited.add(next);
                    }
                }
            }

            // 增加步数
            step++;
        }

        // 如果穷举完还没有找到目标，则说明无法解锁
        return -1;
    }
}
```
